    switchMaterial

    • switchMaterial(
          colorConfig: ColorConfig,
          textureBasePath?: string,
      ): Promise<void>

      切换材质

      参数

      • colorConfig: ColorConfig

        颜色配置对象

      • 可选textureBasePath: string

        纹理基础路径（可选），如果材质配置中的纹理路径是相对路径，则会与此路径拼接

      返回 Promise<void>

      Promise，等待所有纹理加载完成

      示例

      const materialConfig = {
        name: "黑色",
        mats: [
          { name: "面板", map: "black.jpg", normalMap: "normal.jpg", normalScale: [0.8, -0.8] },
          { name: "背面", color: "#000000", emissive: "#ffffff" }
        ]
      };
      await model.switchMaterial(materialConfig, "/public/models/D5103/textures/");

    switchSKU

    • switchSKU(sku_id: string): Promise<void>

      切换SKU

      参数

      • sku_id: string

        应用SKU的材质配置，可选地覆盖渲染参数、背景、环境贴图

      返回 Promise<void>

      示例

      model.switchSKU('sku_black_001');
